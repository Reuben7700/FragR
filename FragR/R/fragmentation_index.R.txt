#' Compute Fragmentation Index for Land Cover
#'
#' @param raster_layer A raster object with binary land cover (1 = forest, 0 = non-forest).
#' @return Numeric fragmentation index (higher = more fragmented).
#' @examples
#' lc <- raster::raster(matrix(sample(c(0,1), 100, replace=TRUE), 10, 10))
#' frag <- fragmentation_index(lc)
#' print(frag)
#' plot(lc, main = paste("Fragmentation Index:", round(frag, 2)))
#' @export
fragmentation_index <- function(raster_layer) {
  if (!inherits(raster_layer, "RasterLayer")) {
    stop("Input must be a RasterLayer.")
  }

  clumps <- raster::clump(raster_layer)

  patch_sizes <- table(raster::values(clumps))

  patch_sizes <- patch_sizes[names(patch_sizes) != "NA"]

  frag_index <- sd(patch_sizes) / mean(patch_sizes)

  plot(clumps, main = paste("Fragmentation Index:", round(frag_index, 2)),
       col = rainbow(length(unique(raster::values(clumps)))))
  
  return(frag_index)
}
